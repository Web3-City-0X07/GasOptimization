# +1 的四种写法之间的燃气差距

### 源码

注意这个代码不小心使用了0.8.1编译器，如果使用其他版本编译，燃气消耗会有些不一样

```js

pragma solidity ^0.8.19;

contract Case_101_A {
    uint256 public number;

    function inc() public returns (uint256) {
        number += 1; 
        return number;
    }
}

contract Case_101_B {
    uint256 public number;
    function inc() public returns (uint256) {
        number = number + 1; 
        return number;
    }
}

contract Case_101_C {
    uint256 public number;
    function inc() public returns (uint256) {
        return number++;
    }
}

contract Case_101_D {
    uint256 public number;
    function inc() public returns (uint256) {
        return ++number;
    }
}

contract Case_101_E {
    uint256 public number;
    function inc() public returns (uint256) {
        unchecked { return ++number; }
    }
}
```
